Windows CMD local process stopping

## Stopping all processes from selected folder:

# Set target path $path = "c:\Path" $processes = Get-Process | Where-Object { $_.Path -like "$path\*" } foreach ($process in $processes) { Write-Host "Stopping process" ($process.Name) " with ID " ($process.Id) "." Stop-Process -Id $Process.Id -Force } Write-Host "All processes from" $path "stopped "
 

## Get remote server user sessions:

quser /server:IP_ADDRESS_SERVER
 

or

qwinsta /server:IP_ADDRESS_SERVER
 

## Log Off remote server user by his session ID:

logoff ... /server:IP_ADDRESS_SERVER
