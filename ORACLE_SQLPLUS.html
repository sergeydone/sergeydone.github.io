### Oracle SQLPLUS commands

 

## SQLPLUS 

	sqlplus "/as sysdba ";
 

## STARTING ORACLE 

	startup;
 

## STARTING ORACLE TO MOUNT STAGE

	startup mount;
 

## STOPPING ORACLE

	shutdown immediate;
 

## ORACLE LISTENER STARTING 

	lsnrctl start;
 

## ORACLE LISTENER STOPPING 

	lsnrctl stop;
