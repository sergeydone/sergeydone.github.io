Postgresql - backup and restore database without data

# 1. Back up only the schema (structure) of a PostgreSQL database without including any data.

pg_dump -U username -h host -p port --schema-only dbname > db_structure.sql

# 2. Create a new empty database (if needed)

createdb -U username -h host -p port newdbname

# 3. Restore the schema dump into new database

psql -U username -h host -p port -d newdbname -f db_structure.sql
