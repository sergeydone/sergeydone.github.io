## Oracle SQL Developer tricks

 

# Для редактирования больших представлений (View) можно воспользоваться следующим методом:

# 1 – Выбрать нужное представление (View)
 
# 2 – Перейти во вкладку «SQL» свойств представления (View)
 
# 3 – Выделить и скопировать ВСЕ содержимое редактируемого представлния (View)
 
# 4 – Перейти во вкладку PL SQL запроса в ПО «Oracle SQL Developer»
 
# 5 – Вставить скопированное содержимое ВСЕГО текста представления (View)
 
# 6 – Добавить/изменить запросы в составе представления (View)
 
# 7 – Выполнить PL SQL - код
 
# Представление (View) изменено.
 

# Из-за особенностей устаревшей версии Oracle 9i выполнение операции с JOB через актуальное ПО Oracle SQL Developer невозможно. Для операций с JOB Oracle 9i применимы PL SQL процедуры. Примеры использования:

# Просмотр имеющихся JOB Oracle 9i 
select * from user_jobs where what like ('%MAIL%')
 
 
# Создание нового JOB Oracle 9i 
declare

x integer;

begin

  sys.dbms_job.submit(job => x,

                      what => 'CHEME_NAME.PKG_NAME.PROCEDURE_NAME;',

                      next_date => to_date('02-02-2000 00:52:47', 'dd-mm-yyyy hh24:mi:ss'),

                      interval => 'SYSDATE + 1/(24*60)');

  commit;

end;
 
 
# Починка сломанного JOB Oracle 9i 
begin

  dbms_job.broken(job => 111, broken => FALSE, next_date => to_date('02-02-2000 00:10:15', 'dd-mm-yyyy hh24:mi:ss'));

end;
 
 
# Запуск JOB Oracle 9i 
begin

  dbms_job.run(job => 111);

end;
 
 
# Остановка JOB Oracle 9i 
begin

  dbms_job.broken(job =>111, broken => TRUE);

end;
 
 
# Удаление остановленного JOB Oracle 9i 
begin

  dbms_job.remove(job => 111);

end;
