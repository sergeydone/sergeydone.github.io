## для сеанса PSQL

# Находим подключенные к БД сеансы:
SELECT * FROM pg_stat_activity WHERE datname = '<database_name>';
 
# Отключаем сессии подключенные к интересующей нас БД с именем <database_name>:
SELECT pg_terminate_backend (pid) FROM pg_stat_activity WHERE pg_stat_activity.datname = '<database_name>';
 
# Удаляем БД:
DROP DATABASE <database_name>
 

## для сеанса postgres:

$ dropdb <database_name>
