# Просмотр параметров настройки сети
ifconfig
 
# [ОПЦИОНАЛЬНО] Альтернативный просмотр параметров настройки сети
ip addr
 
# Просмотр сетевых устройств
nmcli d
 
# Просмотр параметров настройки сети для выбранного сетевого устройства
ip a show [network_device_name]
 
# [ОПЦИОНАЛЬНО] Запуск утилиты для настройки параметров сети
nmtui
 
# [ОПЦИОНАЛЬНО] Перезагрузка сетевого сервиса для применения изменений
service network restart

## [ОПЦИОНАЛЬНО] Настройка firewall

# [ОПЦИОНАЛЬНО] Проверка статуса firewall
firewall-cmd --state
 
# [ОПЦИОНАЛЬНО] Остановка firewall
sudo systemctl stop firewalld
 
# [ОПЦИОНАЛЬНО] Отключение firewall
sudo systemctl disable firewalld
 
# [ОПЦИОНАЛЬНО] Маскирование firewall чтобы его не могли вызвать сервисы
sudo systemctl mask --now firewalld
